swipe事件测试

触发事件
	
var $box = $( '#box' ),
	$btn = $( '#btn' ),
	log = $( 'log' )[0];

$box.on( 'swipeleft', function( e ){
	log.innerHTML += (e.type+"<br/>");
});
$box.on( 'swiperight', function( e ){
	log.innerHTML += (e.type+"<br/>");
});
$box.on( 'swipeup', function( e ){
	log.innerHTML += (e.type+"<br/>");
});
$box.on( 'swipedown', function( e ){
	log.innerHTML += (e.type+"<br/>");
});

$btn.on( 'click', function(){
	$box.off( 'swipeleft');
	$box.off( 'swiperight');
	$box.off( 'swipeup');
	$box.off( 'swipedown');
});
